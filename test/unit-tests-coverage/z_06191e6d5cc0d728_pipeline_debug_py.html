106def _escape_html(value: Any) -> str: 

107 """Escape a value for safe HTML display.""" 

108 if value is None: 

109 return '<span class="null">null</span>' 

110 if isinstance(value, bool): 

111 return f'<span class="bool">{str(value).lower()}</span>' 

112 if isinstance(value, (int, float)): 

113 return f'<span class="number">{value}</span>' 

114 if isinstance(value, list): 

115 if not value: 

116 return '<span class="empty">[]</span>' 

117 items = ", ".join(_escape_html(v) for v in value) 

118 return f"[{items}]" 

119 if isinstance(value, dict): 

120 if not value: 

121 return '<span class="empty">{}</span>' 

122 items = ", ".join(f"{k}: {_escape_html(v)}" for k, v in value.items()) 

123 return f"{{{items}}}" 

124 return html.escape(str(value))

127def _format_segment_html(segment: PathSegmentInfo, index: int) -> str: 

128 """Format a single segment as HTML. 

129 

130 Args: 

131 segment: The segment info 

132 index: Segment index 

133 

134 Returns: 

135 HTML string 

136 """ 

137 status_class = "success" if not segment.errors else "error" 

138 executed_badge = ( 

139 '<span class="badge executed">Executed</span>' 

140 if segment.executed 

141 else '<span class="badge not-executed">Not Executed</span>' 

142 ) 

143 

144 errors_html = "" 

145 if segment.errors: 

146 error_items = "".join( 

147 f"<li>{html.escape(e)}</li>" for e in segment.errors 

148 ) 

149 errors_html = f'<div class="errors"><strong>Errors:</strong><ul>{error_items}</ul></div>' 

150 

151 params_html = "" 

152 if segment.input_parameters: 

153 param_rows = "" 

154 for p in segment.input_parameters: 

155 req_class = "required" if p.required else "optional" 

156 param_rows += f""" 

157 <tr> 

158 <td>{html.escape(p.name)}</td> 

159 <td class="{req_class}">{"required" if p.required else "optional"}</td> 

160 <td>{html.escape(p.source or "")}</td> 

161 <td>{_escape_html(p.value)}</td> 

162 </tr>""" 

163 params_html = f""" 

164 <div class="parameters"> 

165 <strong>Parameters:</strong> 

166 <table> 

167 <tr><th>Name</th><th>Required</th><th>Source</th><th>Value</th></tr> 

168 {param_rows} 

169 </table> 

170 </div>""" 

171 

172 aliases_html = "" 

173 if segment.aliases_involved: 

174 aliases_list = " → ".join(html.escape(a) for a in segment.aliases_involved) 

175 aliases_html = f'<div class="aliases"><strong>Aliases:</strong> {aliases_list}</div>' 

176 

177 intermediate_html = "" 

178 if segment.intermediate_output is not None: 

179 output_preview = str(segment.intermediate_output) 

180 if len(output_preview) > 500: 

181 output_preview = output_preview[:500] + "..." 

182 intermediate_html = f""" 

183 <div class="output"> 

184 <strong>Output:</strong> 

185 <pre>{html.escape(output_preview)}</pre> 

186 </div>""" 

187 

188 return f""" 

189 <div class="segment {status_class}"> 

190 <h3>Segment {index}: {html.escape(segment.segment_text)} {executed_badge}</h3> 

191 <table class="info"> 

192 <tr><th>Type:</th><td>{html.escape(segment.segment_type)}</td></tr> 

193 <tr><th>Resolution:</th><td>{html.escape(segment.resolution_type)}</td></tr> 

194 <tr><th>Server Name:</th><td>{_escape_html(segment.server_name)}</td></tr> 

195 <tr><th>Language:</th><td>{_escape_html(segment.implementation_language)}</td></tr> 

196 <tr><th>Supports Chaining:</th><td>{_escape_html(segment.supports_chaining)}</td></tr> 

197 <tr><th>Valid CID:</th><td>{_escape_html(segment.is_valid_cid)}</td></tr> 

198 <tr><th>Definition CID:</th><td>{_escape_html(segment.server_definition_cid)}</td></tr> 

199 <tr><th>Invocation CID:</th><td>{_escape_html(segment.server_invocation_cid)}</td></tr> 

200 </table> 

201 {aliases_html} 

202 {params_html} 

203 {errors_html} 

204 {intermediate_html} 

205 </div>"""

292def _format_segment_text(segment: PathSegmentInfo, index: int) -> str: 

293 """Format a single segment as plain text. 

294 

295 Args: 

296 segment: The segment info 

297 index: Segment index 

298 

299 Returns: 

300 Plain text string 

301 """ 

302 lines = [ 

303 f"=== Segment {index}: {segment.segment_text} ===", 

304 f"Type: {segment.segment_type}", 

305 f"Resolution: {segment.resolution_type}", 

306 f"Executed: {segment.executed}", 

307 ] 

308 

309 if segment.server_name: 

310 lines.append(f"Server Name: {segment.server_name}") 

311 if segment.implementation_language: 

312 lines.append(f"Language: {segment.implementation_language}") 

313 if segment.server_definition_cid: 

314 lines.append(f"Definition CID: {segment.server_definition_cid}") 

315 

316 lines.append(f"Supports Chaining: {segment.supports_chaining}") 

317 lines.append(f"Valid CID: {segment.is_valid_cid}") 

318 

319 if segment.cid_validation_error: 

320 lines.append(f"CID Error: {segment.cid_validation_error}") 

321 

322 if segment.aliases_involved: 

323 lines.append(f"Aliases: {' -> '.join(segment.aliases_involved)}") 

324 

325 if segment.input_parameters: 

326 lines.append("Parameters:") 

327 for p in segment.input_parameters: 

328 req = "required" if p.required else "optional" 

329 source = p.source or "-" 

330 value = str(p.value) if p.value is not None else "-" 

331 lines.append(f" - {p.name}: {value} ({req}, from {source})") 

332 

333 if segment.errors: 

334 lines.append("Errors:") 

335 for e in segment.errors: 

336 lines.append(f" ! {e}") 

337 

338 if segment.intermediate_output is not None: 

339 output_str = str(segment.intermediate_output) 

340 if len(output_str) > 200: 

341 output_str = output_str[:200] + "..." 

342 lines.append(f"Output: {output_str}") 

343 

344 if segment.server_invocation_cid: 

345 lines.append(f"Invocation CID: {segment.server_invocation_cid}") 

346 

347 return "\n".join(lines)

350def format_debug_text(result: PipelineExecutionResult) -> Response: 

351 """Format pipeline result as plain text response. 

352 

353 Args: 

354 result: The pipeline execution result 

355 

356 Returns: 

357 Flask Response with plain text content 

358 """ 

359 lines = [ 

360 "PIPELINE DEBUG", 

361 "=" * 60, 

362 f"Status: {'SUCCESS' if result.success else 'FAILED'}", 

363 ] 

364 

365 if result.error_message: 

366 lines.append(f"Error: {result.error_message}") 

367 

368 lines.append(f"Segments: {len(result.segments)}") 

369 lines.append("") 

370 

371 for i, seg in enumerate(result.segments): 

372 lines.append(_format_segment_text(seg, i)) 

373 lines.append("") 

374 

375 if result.final_output is not None: 

376 lines.append("=" * 60) 

377 lines.append("FINAL OUTPUT") 

378 lines.append("-" * 60) 

379 output_str = str(result.final_output) 

380 if len(output_str) > 500: 

381 output_str = output_str[:500] + "..." 

382 lines.append(output_str) 

383 lines.append(f"Content-Type: {result.final_content_type}") 

384 

385 return Response("\n".join(lines), mimetype="text/plain")

388def format_debug_response( 

389 result: PipelineExecutionResult, output_format: Optional[str] = None 

390) -> Response: 

391 """Format the pipeline execution result as a debug response. 

392 

393 Respects the output format parameter: 

394 - "json" or None -> JSON response (default) 

395 - "html" -> HTML formatted response 

396 - "txt" or "text" -> Plain text response 

397 

398 Args: 

399 result: The pipeline execution result 

400 output_format: The desired output format (json, html, txt) 

401 

402 Returns: 

403 Flask Response with appropriate content type 

404 """ 

405 if output_format is None: 

406 output_format = "json" 

407 

408 fmt = output_format.lower() 

409 

410 if fmt == "html": 

411 return format_debug_html(result) 

412 if fmt in ("txt", "text"): 

413 return format_debug_text(result) 

414 

415 # Default to JSON 

416 return format_debug_json(result)
