100def validate_limit( 

101 limit: int, 

102 max_allowed: int, 

103 field_name: str = "limit", 

104 min_value: int = 1, 

105) -> Optional[Dict[str, Any]]: 

106 """Validate that a limit/pagination parameter is within acceptable bounds. 

107 

108 This function ensures that pagination limits are: 

109 1. Positive (>= min_value, typically 1) 

110 2. Not exceeding the service's documented maximum 

111 

112 Args: 

113 limit: The limit value to validate 

114 max_allowed: Maximum limit allowed by the external API 

115 field_name: Name of the parameter being validated (for error messages) 

116 min_value: Minimum acceptable value (default: 1) 

117 

118 Returns: 

119 None if validation passes, otherwise a validation error dict 

120 

121 Examples: 

122 >>> # GitHub API with max 100 per page 

123 >>> error = validate_limit(50, GITHUB_MAX_PER_PAGE, "per_page") 

124 >>> error is None 

125 True 

126 

127 >>> # Exceeds GitHub's limit 

128 >>> error = validate_limit(200, GITHUB_MAX_PER_PAGE, "per_page") 

129 >>> error["output"]["error"]["message"] 

130 'per_page exceeds maximum allowed value of 100' 

131 

132 >>> # Negative limit 

133 >>> error = validate_limit(-5, GITHUB_MAX_PER_PAGE, "per_page") 

134 >>> error["output"]["error"]["message"] 

135 'per_page must be at least 1' 

136 

137 Rationale for enforcement: 

138 - Prevents API errors from exceeding service limits 

139 - Provides early feedback to users about constraints 

140 - Documented in preview/dry-run responses for transparency 

141 - Based on official API documentation for each service 

142 """ 

143 if limit < min_value: 

144 return validation_error( 

145 f"{field_name} must be at least {min_value}", 

146 field=field_name, 

147 details={ 

148 "provided": limit, 

149 "minimum": min_value, 

150 "maximum": max_allowed, 

151 "rationale": f"Service requires {field_name} >= {min_value}", 

152 }, 

153 ) 

154 

155 if limit > max_allowed: 

156 return validation_error( 

157 f"{field_name} exceeds maximum allowed value of {max_allowed}", 

158 field=field_name, 

159 details={ 

160 "provided": limit, 

161 "maximum": max_allowed, 

162 "minimum": min_value, 

163 "rationale": f"External API enforces {field_name} <= {max_allowed}", 

164 }, 

165 ) 

166 

167 return None

170def validate_pagination_params( 

171 limit: Optional[int] = None, 

172 max_allowed: int = DEFAULT_MAX_LIMIT, 

173 offset: Optional[int] = None, 

174 page: Optional[int] = None, 

175) -> Optional[Dict[str, Any]]: 

176 """Validate common pagination parameters together. 

177 

178 Args: 

179 limit: Number of items to return (if provided) 

180 max_allowed: Maximum limit for this service 

181 offset: Starting offset (if provided) 

182 page: Page number (if provided) 

183 

184 Returns: 

185 None if all validations pass, otherwise the first validation error 

186 

187 Examples: 

188 >>> # Valid pagination 

189 >>> error = validate_pagination_params(limit=50, offset=0, page=1) 

190 >>> error is None 

191 True 

192 

193 >>> # Invalid limit 

194 >>> error = validate_pagination_params(limit=5000, max_allowed=100) 

195 >>> "exceeds maximum" in error["output"]["error"]["message"] 

196 True 

197 """ 

198 if limit is not None: 

199 if error := validate_limit(limit, max_allowed, "limit"): 

200 return error 

201 

202 if offset is not None and offset < 0: 

203 return validation_error( 

204 "offset must be non-negative", 

205 field="offset", 

206 details={"provided": offset, "minimum": 0}, 

207 ) 

208 

209 if page is not None and page < 1: 

210 return validation_error( 

211 "page must be at least 1", 

212 field="page", 

213 details={"provided": page, "minimum": 1}, 

214 ) 

215 

216 return None

219def get_limit_info(limit: int, max_allowed: int, field_name: str = "limit") -> Dict[str, Any]: 

220 """Get information about a limit value for inclusion in previews/docs. 

221 

222 This provides transparency about limit constraints in dry-run responses. 

223 

224 Args: 

225 limit: The limit value being used 

226 max_allowed: Maximum allowed by the API 

227 field_name: Name of the limit parameter 

228 

229 Returns: 

230 Dictionary with limit information for documentation 

231 

232 Example: 

233 >>> info = get_limit_info(50, 100, "per_page") 

234 >>> info["current"] 

235 50 

236 >>> info["maximum"] 

237 100 

238 """ 

239 return { 

240 "parameter": field_name, 

241 "current": limit, 

242 "maximum": max_allowed, 

243 "status": "valid" if limit <= max_allowed else "exceeds_maximum", 

244 "constraint_source": "external_api_documentation", 

245 }
