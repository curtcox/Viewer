9@dataclass(frozen=True) 

10class CommandInfo: 

11 """Details about a common bash command available as a server.""" 

12 

13 name: str 

14 description: str 

15 role: str # Source | Transform | Dual 

16 

17 @property 

18 def safe_for_readonly(self) -> bool: 

19 """Commands marked as Dual can mutate system state and stay out of readonly boots.""" 

20 

21 return self.role.lower() != "dual"

24def _command(name: str, description: str, role: str) -> CommandInfo: 

25 return CommandInfo(name=name, description=description, role=role)

130def group_commands_for_readonly(commands: Iterable[CommandInfo]) -> list[CommandInfo]: 

131 """Return commands safe to include in the readonly boot image.""" 

132 

133 return [command for command in commands if command.safe_for_readonly]
